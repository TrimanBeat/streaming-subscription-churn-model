  1. Limpieza ligera y variables derivadas La variable signup_date viene como número negativo de días; la convertimos en una fecha relativa, tomando como referencia la fecha de hoy, y creamos una variable de antigüedad en años. También creamos algunas variables de interés para producto.
Code 
train_clean <- train %>%
  mutate(
    # signup_date: días desde hoy (negativo -> pasado)
    signup_date = as.numeric(signup_date),
    signup_date = as_date(signup_date, origin = Sys.Date()),
    tenure_days = as.numeric(Sys.Date() - signup_date),
    tenure_years = tenure_days / 365,
    churned = as.factor(churned),
    subscription_type = as.factor(subscription_type),
    payment_plan = as.factor(payment_plan),
    payment_method = as.factor(payment_method),
    customer_service_inquiries = factor(
      customer_service_inquiries,
      levels = c("Low", "Medium", "High"),
      ordered = TRUE
    )
  )

train_cohorts <- train_clean %>%
  mutate(
    signup_year = year(signup_date),
    tenure_bucket = case_when(
      tenure_years < 1 ~ "<1 año",
      tenure_years < 2 ~ "1-2 años",
      tenure_years < 3 ~ "2-3 años",
      TRUE ~ ">=3 años"
    )
  )
